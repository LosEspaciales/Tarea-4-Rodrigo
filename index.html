import geopandas as gpd
import os
import matplotlib.pyplot as plt
from shapely.validation import explain_validity, make_valid
In[]:
ruta = os.path.join("data", "kontur_boundaries_AU_20230628.gpkg")
australia_gdf = gpd.read_file(ruta)
In[]:
# Obtener centroides
centroides = australia_gdf.centroid

# Definir coordenadas del centro (aproximadamente)
centroidX = centroides.x.mean()
centroidY = centroides.y.mean()

# Crear subconjuntos de polgonos por regin
Australia_N = australia_gdf.cx[:, centroidY:]
Australia_S = australia_gdf.cx[:, :centroidY]
Australia_E = australia_gdf.cx[centroidX:, :]
Australia_W = australia_gdf.cx[:centroidX, :]

# Crear convex hulls para cada regin
Australia_N_hull = Australia_N.dissolve().convex_hull
Australia_S_hull = Australia_S.dissolve().convex_hull
Australia_E_hull = Australia_E.dissolve().convex_hull
Australia_W_hull = Australia_W.dissolve().convex_hull
In[]:
# Encontrar y visualizar geometras invlidas
invalid_geometries = australia_gdf[~australia_gdf.is_valid]
invalid_geometries.plot()

# Explicar la validez de las geometras invlidas
invalid_explanations = invalid_geometries.geometry.apply(explain_validity)

# Corregir geometras invlidas
australia_gdf_valid = australia_gdf.copy()
australia_gdf_valid['geometry'] = australia_gdf_valid['geometry'].apply(lambda geom: make_valid(geom) if not geom.is_valid else geom)

# Verificar nuevamente geometras invlidas
remaining_invalid_geometries = australia_gdf_valid[~australia_gdf_valid.is_valid]
print(f"Geometras invlidas restantes: {remaining_invalid_geometries.shape[0]}")
In[]:
# Crear un GeoDataFrame para los convex hulls
hulls_gdf = gpd.GeoDataFrame(geometry=[Australia_N_hull, Australia_S_hull, Australia_E_hull, Australia_W_hull], crs=australia_gdf.crs)

# Graficar el pas con los convex hulls
base = australia_gdf.plot(color='white', edgecolor='black')
hulls_gdf.plot(ax=base, color=['grey', 'orange', 'green', 'red'], alpha=0.4)
plt.show()
